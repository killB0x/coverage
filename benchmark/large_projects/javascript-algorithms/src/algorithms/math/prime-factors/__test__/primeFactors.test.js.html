"use strict";
 
var _primeFactors = require("../primeFactors");
/**
 * Calculates the error between exact and approximate prime factor counts.
 * @param {number} exactCount
 * @param {number} approximateCount
 * @returns {number} - approximation error (percentage).
 */
function approximationError(exactCount, approximateCount) {
  return Math.abs((exactCount - approximateCount) / exactCount) * 100;
}
describe('primeFactors', () => {
  it('should find prime factors', () => {
    expect((0, _primeFactors.primeFactors)(1)).toEqual([]);
    expect((0, _primeFactors.primeFactors)(2)).toEqual([2]);
    expect((0, _primeFactors.primeFactors)(3)).toEqual([3]);
    expect((0, _primeFactors.primeFactors)(4)).toEqual([2, 2]);
    expect((0, _primeFactors.primeFactors)(14)).toEqual([2, 7]);
    expect((0, _primeFactors.primeFactors)(40)).toEqual([2, 2, 2, 5]);
    expect((0, _primeFactors.primeFactors)(54)).toEqual([2, 3, 3, 3]);
    expect((0, _primeFactors.primeFactors)(100)).toEqual([2, 2, 5, 5]);
    expect((0, _primeFactors.primeFactors)(156)).toEqual([2, 2, 3, 13]);
    expect((0, _primeFactors.primeFactors)(273)).toEqual([3, 7, 13]);
    expect((0, _primeFactors.primeFactors)(300)).toEqual([2, 2, 3, 5, 5]);
    expect((0, _primeFactors.primeFactors)(980)).toEqual([2, 2, 5, 7, 7]);
    expect((0, _primeFactors.primeFactors)(1000)).toEqual([2, 2, 2, 5, 5, 5]);
    expect((0, _primeFactors.primeFactors)(52734)).toEqual([2, 3, 11, 17, 47]);
    expect((0, _primeFactors.primeFactors)(343434)).toEqual([2, 3, 7, 13, 17, 37]);
    expect((0, _primeFactors.primeFactors)(456745)).toEqual([5, 167, 547]);
    expect((0, _primeFactors.primeFactors)(510510)).toEqual([2, 3, 5, 7, 11, 13, 17]);
    expect((0, _primeFactors.primeFactors)(8735463)).toEqual([3, 3, 11, 88237]);
    expect((0, _primeFactors.primeFactors)(873452453)).toEqual([149, 1637, 3581]);
  });
  it('should give approximate prime factors count using Hardy-Ramanujan theorem', () => {
    expect((0, _primeFactors.hardyRamanujan)(2)).toBeCloseTo(-0.366, 2);
    expect((0, _primeFactors.hardyRamanujan)(4)).toBeCloseTo(0.326, 2);
    expect((0, _primeFactors.hardyRamanujan)(40)).toBeCloseTo(1.305, 2);
    expect((0, _primeFactors.hardyRamanujan)(156)).toBeCloseTo(1.6193, 2);
    expect((0, _primeFactors.hardyRamanujan)(980)).toBeCloseTo(1.929, 2);
    expect((0, _primeFactors.hardyRamanujan)(52734)).toBeCloseTo(2.386, 2);
    expect((0, _primeFactors.hardyRamanujan)(343434)).toBeCloseTo(2.545, 2);
    expect((0, _primeFactors.hardyRamanujan)(456745)).toBeCloseTo(2.567, 2);
    expect((0, _primeFactors.hardyRamanujan)(510510)).toBeCloseTo(2.575, 2);
    expect((0, _primeFactors.hardyRamanujan)(8735463)).toBeCloseTo(2.771, 2);
    expect((0, _primeFactors.hardyRamanujan)(873452453)).toBeCloseTo(3.024, 2);
  });
  it('should give correct deviation between exact and approx counts', () => {
    expect(approximationError((0, _primeFactors.primeFactors)(2).length, (0, _primeFactors.hardyRamanujan)(2))).toBeCloseTo(136.651, 2);
    expect(approximationError((0, _primeFactors.primeFactors)(4).length, (0, _primeFactors.hardyRamanujan)(2))).toBeCloseTo(118.325, 2);
    expect(approximationError((0, _primeFactors.primeFactors)(40).length, (0, _primeFactors.hardyRamanujan)(2))).toBeCloseTo(109.162, 2);
    expect(approximationError((0, _primeFactors.primeFactors)(156).length, (0, _primeFactors.hardyRamanujan)(2))).toBeCloseTo(109.162, 2);
    expect(approximationError((0, _primeFactors.primeFactors)(980).length, (0, _primeFactors.hardyRamanujan)(2))).toBeCloseTo(107.330, 2);
    expect(approximationError((0, _primeFactors.primeFactors)(52734).length, (0, _primeFactors.hardyRamanujan)(52734))).toBeCloseTo(52.274, 2);
    expect(approximationError((0, _primeFactors.primeFactors)(343434).length, (0, _primeFactors.hardyRamanujan)(343434))).toBeCloseTo(57.578, 2);
